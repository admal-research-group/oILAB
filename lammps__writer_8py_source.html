4def compute_triclinic_box(ell1, ell2, zlo=-20.0, zhi=20.0):
5 """
6 Convert two 2D Cartesian box vectors into LAMMPS triclinic box parameters.
7
8 Assumes the user has already rotated the system, if desired.
9
10 Parameters
11 ----------
12 ell1, ell2 : array-like
13 2D Cartesian simulation cell vectors.
14 zlo, zhi : float
15 Lower and upper z bounds.
16
17 Returns
18 -------
19 dict
20 Dictionary with xlo, xhi, ylo, yhi, zlo, zhi, xy, xz, yz.
21 """
22 ell1 = np.asarray(ell1, dtype=float).reshape(2)
23 ell2 = np.asarray(ell2, dtype=float).reshape(2)
24
25 # Standard 2D triclinic reduction used by LAMMPS
26 lx = np.linalg.norm(ell1)
27 if lx <= 0.0:
28 raise ValueError("ell1 has zero length.")
29
30 ex = ell1 / lx
31 xy = np.dot(ell2, ex)
32
33 ell2_perp = ell2 - xy * ex
34 ly = np.linalg.norm(ell2_perp)
35 if ly <= 0.0:
36 raise ValueError("ell1 and ell2 are collinear.")
37
38 return {
39 "xlo": 0.0,
40 "xhi": float(lx),
41 "ylo": 0.0,
42 "yhi": float(ly),
43 "zlo": float(zlo),
44 "zhi": float(zhi),
45 "xy": float(xy),
46 "xz": 0.0,
47 "yz": 0.0,
48 }

63 Write a LAMMPS data file.
64
65 Parameters

67 filename : str
68 Output file name.

71 atom_types : list[int]
72 Atom types.
73 charges : list[float]
74 Atomic charges. Used only for atom_style='full'.
75 ell1, ell2 : array-like
76 2D Cartesian simulation cell vectors.
77 atom_style : str
78 'atomic' or 'full'.
79 z_padding : float
80 Half-width of z box: [ -z_padding, z_padding ].
81 comment : str
82 Header comment.
83 """
84 if len(positions) != len(atom_types):
85 raise ValueError("positions and atom_types must have the same length.")
86
87 if atom_style == "full" and len(positions) != len(charges):
88 raise ValueError("positions and charges must have the same length for atom_style='full'.")
89
90 box = compute_triclinic_box(ell1, ell2, zlo=-z_padding, zhi=z_padding)
91
92 n_atoms = len(positions)
93 n_atom_types = max(atom_types)
94
95 with open(filename, "w", encoding="utf-8") as f:

99
100 f.write(f"{box['xlo']:.16f} {box['xhi']:.16f} xlo xhi\n")
101 f.write(f"{box['ylo']:.16f} {box['yhi']:.16f} ylo yhi\n")
102 f.write(f"{box['zlo']:.16f} {box['zhi']:.16f} zlo zhi\n")
103 f.write(f"{box['xy']:.16f} {box['xz']:.16f} {box['yz']:.16f} xy xz yz\n\n")
104
105 if atom_style == "atomic":
106 f.write("Atoms # atomic\n\n")
107 for i, ((x, y, z), atom_type) in enumerate(zip(positions, atom_types), start=1):
108 f.write(f"{i:d} {atom_type:d} {x:.16f} {y:.16f} {z:.16f}\n")
109
110 elif atom_style == "full":
111 f.write("Atoms # full\n\n")
112 for i, ((x, y, z), atom_type, charge) in enumerate(
113 zip(positions, atom_types, charges), start=1
114 ):
115 mol_id = 1

117 f"{i:d} {mol_id:d} {atom_type:d} "

120 else:
