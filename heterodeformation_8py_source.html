18def as_fraction(x):
19 """Convert x to Fraction unless it is already a Fraction."""
20 return x if isinstance(x, Fraction) else Fraction(x)

23def gcd_fraction(x, y):
24 """Greatest common divisor of two rational numbers.
25
26 If x = a/b and y = c/d in lowest terms, define
27 gcd(x, y) = gcd(ad, cb)/(bd).
28 """
29 x = as_fraction(x)
30 y = as_fraction(y)
31 a, b = x.numerator, x.denominator
32 c, d = y.numerator, y.denominator
33 return Fraction(gcd(a * d, c * b), b * d)

36def gcd_fraction_list(vals):
37 """Greatest common divisor of a list of rational numbers."""
38 vals = [as_fraction(v) for v in vals]
39 nz = [v for v in vals if v != 0]
40 if len(nz) == 0:
41 return Fraction(1, 1)
42 return reduce(gcd_fraction, nz)

45def dot_frac(u, v):
46 """Exact dot product of two rational vectors."""
47 return sum(as_fraction(ui) * as_fraction(vi) for ui, vi in zip(u, v))

50def outer_frac(u, v):
51 """Exact outer product of two rational 2-vectors."""
52 return np.array(
53 [[as_fraction(u[i]) * as_fraction(v[j]) for j in range(2)] for i in range(2)],

58def frac_abs(x):
59 """Absolute value for Fraction."""

64def primitive_orthogonal(l):
65 """Smallest reciprocal vector orthogonal to l.
66
67 If l = (l1, l2), then
68 mbar = (l2, -l1) / gcd(l1, l2).
69 """
70 l1 = as_fraction(l[0])
71 l2 = as_fraction(l[1])
72 g = gcd_fraction_list([l1, l2])
73 return np.array([l2 / g, -l1 / g], dtype=object)

77 """Convert a matrix with Fraction entries to a NumPy float array."""

84def inv2_frac(M):
85 """Exact inverse of a 2x2 matrix with Fraction entries."""
86 a, b = as_fraction(M[0, 0]), as_fraction(M[0, 1])
87 c, d = as_fraction(M[1, 0]), as_fraction(M[1, 1])
88 det = a * d - b * c
89 if det == 0:
90 raise ValueError("Q is singular, so Q^{-1} does not exist.")
91 return np.array([[d / det, -b / det], [-c / det, a / det]], dtype=object)

112# DEFAULT EXAMPLE:
113# 2D simple twist in bilayer graphene (Fig. 7 in the manuscript)
114# Replace A, b1u, b2u, b3u, l1u, l2u with any case listed in README.
115
116# Structure matrix of the 2D triangular Bravais lattice
117# Graphene: a = 2.46 A -> A = (a/2) [[0, -sqrt(3)], [2, -1]]
118# MoS2: a = 3.19702 A -> same normalized matrix scaled by a
119A = (2.46 / 2.0) * np.array([[0.0, -np.sqrt(3.0)],
120 [2.0, -1.0]], dtype=float)
121
122# Burgers vectors in lattice coordinates
123b1u = np.array([Fraction(-1, 3), Fraction(-2, 3)], dtype=object)
124b2u = np.array([Fraction(-1, 3), Fraction(1, 3)], dtype=object)
125b3u = np.array([Fraction(2, 3), Fraction(1, 3)], dtype=object)
126
127# Line vectors in lattice coordinates
128l1u = np.array([Fraction(-110, 1), Fraction(-221, 1)], dtype=object)
129l2u = np.array([Fraction(-111, 1), Fraction(110, 1)], dtype=object)
130
131# beta = 1 for 1D networks and 2D triangular networks,
132# beta = 3 for 2D honeycomb networks.
133# Set this explicitly after choosing an example from the README.
134beta = Fraction(1, 1)

137# ------------------------------------------------------------
138# Main calculation
139# ------------------------------------------------------------
140l3u = -(l1u + l2u)
141B = [b1u, b2u, b3u]
142L = [l1u, l2u, l3u]

144print("Cartesian Burgers vectors:")

152mbar = [primitive_orthogonal(li) for li in L]
153z1 = frac_abs(dot_frac(L[1], mbar[0]))
154z2 = frac_abs(dot_frac(L[2], mbar[1]))
155z3 = frac_abs(dot_frac(L[0], mbar[2]))

174Finv = A @ Qf @ Ainv
