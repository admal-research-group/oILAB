68 template <int dim>
69 RationalReciprocalLatticeDirection<dim> RationalReciprocalLatticeDirection<dim>::operator+(const RationalReciprocalLatticeDirection<dim> &other) const
70 {
71 assert(&dir.lattice == &other.dir.lattice && "Rational Lattice Vector Type belong to different Lattices.");
72 const VectorDimI temp(rat.n * other.rat.d * dir.reciprocalLatticeVector() + other.rat.n * rat.d * other.dir.reciprocalLatticeVector());
73 const IntScalarType gcd(IntegerMath<IntScalarType>::gcd(temp));
74 const ReciprocalLatticeVector<dim> v((temp / gcd).eval(), dir.lattice);
75 return RationalReciprocalLatticeDirection<dim>(Rational(gcd, rat.d * other.rat.d), ReciprocalLatticeDirection<dim>(v));
76 }
